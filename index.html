Git-Commands

  1. git clone "URL"
  2. git add ./filename
  3. git status
  4. git commit -m "meassage"
  5. git push
  6. git pull
  7. git commit -am "meassage"
  8. git reset -hard < commit hash >
  9. git reset -hard origin/master

  10. git branch
  11. git checkout -b "New branch name"
  12. git checkout "branch name"
  13. git merge "branch name"
  14. git pull origin main
  15. git push origin --delete [branch name]
